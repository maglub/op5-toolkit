#
# 1) list instances in ec2
# 2) use awk to format a ./monitor.rb command
#                   ===== 1 ====                ====== 2 =====
../../nagios-cloudwatch/check_cloudwatch.rb --ec2 --list-instances | awk '{print "./monitor.rb -t host --create --options='\''host_name="$2",alias="$2",address=127.0.0.1,_EC2_ID="$4",hostgroups=aws-ec2-instances,_EC2_REGION=us-west-2'\''"}'
